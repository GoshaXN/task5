Для решения Quest 1 необходимо доработать модуль data_module так, чтобы data_module_entry.c компилировался и корректно выполнял нормализацию входных данных. Основные шаги:  1. Подключить необходимые заголовки в data_module_entry.c и добавить ввод n, выделение памяти, вызов input, normalization и output. 2. Реализовать функции ввода/вывода в data_io.c (чтение массива и вывод с форматом %.2f). 3. Реализовать статистические функции в data_stat.c (max, min, mean, variance). 4. Исправить логику нормализации в data_process.c: условие должно быть if (fabs(size) > EPS) – нормализовать, иначе вернуть ошибку. 5. Добавить включение data_stat.h в data_process.c для использования max и min.  Ниже приведены изменённые файлы (полный код).  ---  1. src/data_module/data_module_entry.c  ```c #include <stdio.h> #include <stdlib.h> #include "data_io.h" #include "data_process.h"  void main() {     double *data;     int n;      // Ввод количества элементов     scanf("%d", &n);      // Выделение памяти     data = (double*)malloc(n * sizeof(double));     if (data == NULL) {         printf("ERROR");         return;     }      input(data, n);      if (normalization(data, n))         output(data, n);     else         printf("ERROR");      free(data); } ```  ---  2. src/data_libs/data_io.c  ```c #include "data_io.h" #include <stdio.h>  void input(double *data, int n) {     for (int i = 0; i < n; i++) {         scanf("%lf", &data[i]);     } }  void output(double *data, int n) {     for (int i = 0; i < n; i++) {         printf("%.2f", data[i]);         if (i < n - 1) printf(" ");     }     printf("\n"); } ```  ---  3. src/data_libs/data_stat.c  ```c #include "data_stat.h" #include <math.h>  double max(double *data, int n) {     double max_val = data[0];     for (int i = 1; i < n; i++) {         if (data[i] > max_val) max_val = data[i];     }     return max_val; }  double min(double *data, int n) {     double min_val = data[0];     for (int i = 1; i < n; i++) {         if (data[i] < min_val) min_val = data[i];     }     return min_val; }  double mean(double *data, int n) {     double sum = 0.0;     for (int i = 0; i < n; i++) sum += data[i];     return sum / n; }  double variance(double *data, int n) {     double m = mean(data, n);     double sum_sq = 0.0;     for (int i = 0; i < n; i++) {         double diff = data[i] - m;         sum_sq += diff * diff;     }     return sum_sq / n; } ```  ---  4. src/data_module/data_process.c  ```c #include "data_process.h" #include "data_stat.h"   // для max и min #include <math.h>  int normalization(double *data, int n) {     int result = 1;     double max_value = max(data, n);     double min_value = min(data, n);     double size = max_value - min_value;      // Исправлено условие: нормализация, если диапазон больше EPS     if (fabs(size) > EPS) {         for (int i = 0; i < n; i++) {             data[i] = (data[i] - min_value) / size;         }     } else {         result = 0;     }      return result; } ```  ---  5.
